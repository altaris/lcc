Exponential Moving Average (EMA) callack for pytorch lightning. Code taken and updated from https://colab.research.google.com/github/benihime91/gale/blob/master/nbs/07b_collections.callbacks.ema.ipynb

  2Exponential Moving Average (EMA) callack for pytorch lightning. Code taken and
  3updated from
  4https://colab.research.google.com/github/benihime91/gale/blob/master/nbs/07b_collections.callbacks.ema.ipynb
  5"""

  7from typing import Any, Iterable
  8
  9import pytorch_lightning as pl
 10from timm.utils.model import get_state_dict, unwrap_model
 11from timm.utils.model_ema import ModelEmaV3
 12from torch import Tensor
 13
 14CKPT_KEY = "state_dict_ema"
 15"""
 16Key to use when saving/loading a checkpoint dict.
 17"""
 18
 19
 20class EMACallback(pl.Callback):
 21    """
 22    Model Exponential Moving Average. Empirically it has been found that using
 23    the moving average of the trained parameters of a deep network is better
 24    than using its trained parameters directly.
 25    """
 26
 27    decay: float
 28    ema: ModelEmaV3
 29    use_ema_weights: bool
 30
 31    def __init__(self, decay: float = 0.9999, use_ema_weights: bool = True):
 32        """
 33        Args:
 34            decay (float, optional):
 35            use_ema_weights (bool, optional): If `True`, then the ema parameters
 36                of the network are set onto the original model after training
 37                end.
 38        """
 39        self.decay, self.use_ema_weights = decay, use_ema_weights
 40
 41    def copy_to(
 42        self, shadow_parameters: Iterable[Tensor], parameters: Iterable[Tensor]
 43    ) -> None:
 44        """Copy current parameters into given collection of parameters."""
 45        for s_param, param in zip(shadow_parameters, parameters):
 46            if param.requires_grad:
 47                param.data.copy_(s_param.data)
 48
 49    def on_fit_start(
 50        self, trainer: pl.Trainer, pl_module: pl.LightningModule
 51    ) -> None:
 52        self.ema = ModelEmaV3(pl_module, decay=self.decay, device=None)
 53        super().on_fit_start(trainer, pl_module)
 54
 55    def on_load_checkpoint(
 56        self,
 57        trainer: pl.Trainer,
 58        pl_module: pl.LightningModule,
 59        checkpoint: dict[str, Any],
 60    ) -> None:
 61        if self.ema is not None:
 62            self.ema.module.load_state_dict(checkpoint[CKPT_KEY])
 63        super().on_load_checkpoint(trainer, pl_module, checkpoint)
 64
 65    def on_save_checkpoint(
 66        self,
 67        trainer: pl.Trainer,
 68        pl_module: pl.LightningModule,
 69        checkpoint: dict[str, Any],
 70    ) -> None:
 71        if self.ema is not None:
 72            checkpoint[CKPT_KEY] = get_state_dict(self.ema, unwrap_model)
 73        super().on_save_checkpoint(trainer, pl_module, checkpoint)
 74
 75    def on_train_batch_end(
 76        self,
 77        trainer: pl.Trainer,
 78        pl_module: pl.LightningModule,
 79        *args: Any,
 80        **kwargs: Any,
 81    ) -> None:
 82        self.ema.update(pl_module)
 83        super().on_train_batch_end(trainer, pl_module, *args, **kwargs)
 84
 85    def on_train_end(
 86        self, trainer: pl.Trainer, pl_module: pl.LightningModule
 87    ) -> None:
 88        if self.use_ema_weights:
 89            self.copy_to(self.ema.module.parameters(), pl_module.parameters())
 90        super().on_train_end(trainer, pl_module)
 91
 92    def on_validation_end(
 93        self, trainer: pl.Trainer, pl_module: pl.LightningModule
 94    ) -> None:
 95        self.restore(pl_module.parameters())
 96        super().on_validation_end(trainer, pl_module)
 97
 98    def on_validation_start(
 99        self, trainer: pl.Trainer, pl_module: pl.LightningModule
100    ) -> None:
101        self.store(pl_module.parameters())
102        self.copy_to(self.ema.module.parameters(), pl_module.parameters())
103        super().on_validation_start(trainer, pl_module)
104
105    def restore(self, parameters: Iterable[Tensor]) -> None:
106        """
107        Restore the parameters stored with the `store` method.  Useful to
108        validate the model with EMA parameters without affecting the original
109        optimization process.
110        """
111        for c_param, param in zip(self.collected_params, parameters):
112            param.data.copy_(c_param.data)
113
114    def store(self, parameters: Iterable[Tensor]) -> None:
115        """Save the current parameters for restoring later."""
116        self.collected_params = [param.clone() for param in parameters]
